## [2504 괄호의 값](https://www.acmicpc.net/problem/2504)

:::tip
Clicking the heading will take you to the BOJ problem.

## Solution

```cpp
#include <iostream>
#include <stack>

using namespace std;

int main() {
    string a;
    cin >> a;

    stack<char> s;
    int ans = 0;
    int tmp = 1;

    for (int i = 0; i < a.size(); i++) {
        if (a[i] == '(') {
            s.push(a[i]);
            tmp *= 2;
        } else if (a[i] == '[') {
            s.push(a[i]);
            tmp *= 3;
        } else if (a[i] == ')') {
            if (s.empty() || s.top() != '(') {
                cout << 0;
                return 0;
            }
            if (a[i - 1] == '(') {
                ans += tmp;
            }
            s.pop();
            tmp /= 2;
        } else if (a[i] == ']') {
            if (s.empty() || s.top() != '[') {
                cout << 0;
                return 0;
            }
            if (a[i - 1] == '[') {
                ans += tmp;
            }
            s.pop();
            tmp /= 3;
        }
    }
    if (!s.empty())
        cout << 0;
    else
        cout << ans;
}
```
