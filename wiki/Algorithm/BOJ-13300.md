## Solution

문제에서 입력으로 받는 숫자를 잘 쓰자. 이차원 배열도 잘 쓰자.

```python
n, k = map(int, input().split())
room = [[0] * 7 for _ in range(2)]

for _ in range(n):
    s, y = map(int, input().split())
    room[s][y] += 1

count = 0
for s in range(2):
    for y in range(1, 7):
        if room[s][y] > 0:
            count += (room[s][y] - 1) // k + 1 # math.ceil(n / k)
print(count)
```

```cpp
#include <iostream>

using namespace std;

int main() {
    int n, k;
    int ans = 0;
    int cnt[2][7] = {};

    cin >> n >> k;
    for (int i = 0; i < n; i++) {
        int s, y;
        cin >> s >> y;
        cnt[s][y]++;
    }

    for (int i = 0; i < 2; i++) {
        for (int j = 1; j < 7; j++) {
            ans += cnt[i][j] / k;
            if (cnt[i][j] % k) {
                ans++;
            }
        }
    }
    cout << ans;
}
```
