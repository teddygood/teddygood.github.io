## Solution

스택을 활용해서 푸는 문제였다. 파이썬이 아닌 C++로 풀어봤다. 내일은 파이썬으로도 풀어보고 스택 문제를 더 풀어봐야겠다.

```cpp
#include <iostream>
#include <vector>
#include <stack>

using namespace std;

int main()
{
    int N;
    cin >> N;
    vector<int> num(N);
  
    for (int i = 0; i < N; i++)
    {
        cin >> num[i];
    }

    stack<int> st;
    vector<int> result(N, -1);

    for (int i = N - 1; i >= 0; i--)
    {
        while (!st.empty() && num[i] >= st.top())
        {
            st.pop();
        }
        if (!st.empty())
        {
            result[i] = st.top();
        }
        st.push(num[i]);
    }
    
    for (auto num : result)
    {
        cout << num << " ";
    }
}

```
