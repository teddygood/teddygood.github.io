## Solution

아이디어는 다 괜찮았는데 절댓값을 생각을 못했었다.

```python
a = input()
b = input()

count_a = [0] * 26
count_b = [0] * 26

for c in a:
    count_a[ord(c) - ord('a')] += 1
for c in b:
    count_b[ord(c) - ord('a')] += 1

answer = 0

for i in range(26):
    if count_a[i] >= count_b[i]:
        answer += count_a[i] - count_b[i]
    else:
        answer += count_b[i] - count_a[i]
        
print(answer)
```

```cpp
#include <iostream>

using namespace std;

int main() {
    string s1, s2;
    cin >> s1 >> s2;
    int arr[2][26] = {0};

    for (char i : s1) {
        arr[0][i - 'a']++;
    }
    for (char i : s2) {
        arr[1][i - 'a']++;
    }

    int ans = 0;
    for (int i = 0; i < 26; i++) {
        ans += abs(arr[0][i] - arr[1][i]);
    }
    cout << ans;
}
```
