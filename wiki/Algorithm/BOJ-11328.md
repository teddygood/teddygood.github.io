## Solution

문자열은 `sort()`를 사용할 수 없다. `sorted()`를 사용하자.

```python
n = int(input())

for _ in range(n):
    a, b = input().split()
    if sorted(b) == sorted(a):
        print("Possible")
    else:
        print("Impossible")
```

```cpp
#include <iostream>

using namespace std;

int main() {
    int n;
    cin >> n;

    while(n--) {
        int cnt[26] = {};
        string s1, s2;
        cin >> s1 >> s2;

        for (auto c : s1) {
            cnt[c-'a']++;
        }
        for (auto c : s2) {
            cnt[c-'a']--;
        }

        bool isPossible = true;

        for (int i : cnt) {
            if (i != 0) {
                isPossible = false;
            }
        }
        if(isPossible) cout << "Possible\n";
        else cout << "Impossible\n";
    }

}
```
